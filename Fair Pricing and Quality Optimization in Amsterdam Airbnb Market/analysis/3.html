2.2 Categorical Variables

Important categorical variables are unevenly distributed. Some of the special class present special price distribution correspondingly. Such as the small portion of cancellation_policy, super_strict_60, corresponds to a higher price. These variables embodied a housing type diversity that can influence the price.

Code 
# Figure6
Am_Basic_no_geo = Am_Basic.drop(columns=['geometry'])
selected_data = Am_Basic_no_geo[['price', 'host_response_time', 'cancellation_policy', 'room_type']]
melted_data = selected_data.melt(id_vars=['price'], 
                                 value_vars=['host_response_time', 'cancellation_policy', 'room_type'], 
                                 var_name='Variable', 
                                 value_name='value')
plt.figure(figsize=(12,10))
g = sns.FacetGrid(melted_data, col="Variable", sharex=False, sharey=False, height=4, aspect=1.5)
g.map_dataframe(sns.barplot, x="value", y="price", estimator="mean", 
                color="#aa003A", edgecolor="#aa003A", alpha=0.4)

g.set_axis_labels("", "Price")
g.set_titles(col_template="{col_name}")
g.set_xticklabels(rotation=45, horizontalalignment='right', size=10)
g.set_titles(fontsize=14)

g.fig.suptitle("Housing Price According to Categorical Features", fontsize=20, y=1.05)
g.fig.subplots_adjust(top=0.85)
plt.figtext(0.5, -0.2, "Figure 6, Housing Price According to Categorical Features", 
            wrap=True, horizontalalignment='center', fontsize=12)

plt.show()

2.4 Boolean Variables

We can also see the difference of price in the different classes of Boolean variables.

Code 
Am_Basic_no_geo = Am_Basic.drop(columns=['geometry'])
selected_data = Am_Basic_no_geo[['price', 'a7', 'a53', 'a100']]
melted_data = selected_data.melt(id_vars=['price'], 
                                 var_name='Variable', 
                                 value_name='value')
def label_features(variable):
    return {
        'a7': 'Smoke Detector',
        'a53': 'Dryer',
        'a100': 'Baby Monitor'
    }.get(variable, variable)

melted_data['Variable'] = melted_data['Variable'].apply(label_features)
plt.figure(figsize=(12, 10))

g = sns.FacetGrid(melted_data, col="Variable", sharex=False, sharey=False, height=5, aspect=1.5)
g.map_dataframe(sns.barplot, x="value", y="price", estimator="mean", 
                color="#aa003A", edgecolor="#aa003A", alpha=0.4)

g.set_axis_labels("", "Average Price")
g.set_titles(col_template="{col_name}")
g.set_xticklabels(size=10)
g.set_titles(fontsize=14)

g.fig.suptitle("Important Boolean Features Distribution", fontsize=20, y=0.95)
g.fig.subplots_adjust(top=0.85)

plt.figtext(0.5, -0.05, "Figure 10", 
            wrap=True, horizontalalignment='center', fontsize=12)

plt.show()
